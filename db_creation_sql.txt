CREATE TABLE "users" (
	"Id"	INTEGER,
	"FirstName"	TEXT,
	"LastName"	TEXT NOT NULL,
	"Email"	TEXT NOT NULL,
	PRIMARY KEY("Id" AUTOINCREMENT)
);

CREATE TABLE "roles" (
	"Id"	TEXT,
	"Description"	TEXT,
	PRIMARY KEY("Id")
);

CREATE TABLE "users_roles" (
	"UserId"	INTEGER,
	"RoleId"	TEXT,
	FOREIGN KEY("UserId") REFERENCES "users"("Id") ON UPDATE CASCADE ON DELETE CASCADE,
	FOREIGN KEY("RoleId") REFERENCES "roles"("Id") ON UPDATE CASCADE ON DELETE RESTRICT
);

CREATE TABLE "forms" (
	"Id"	INTEGER,
	"RequestorId"	INTEGER NOT NULL,
	"Status"	INTEGER NOT NULL,
	"Submitted"	INTEGER NOT NULL,
	"Modified"	INTEGER NOT NULL,
	FOREIGN KEY("RequestorId") REFERENCES "users"("Id") ON UPDATE CASCADE ON DELETE NO ACTION,
	PRIMARY KEY("Id")
);

CREATE TABLE "form_items" (
	"Id"	INTEGER,
	"FormId"	INTEGER NOT NULL,
	"Description"	TEXT NOT NULL,
	"Cost"	NUMERIC NOT NULL,
	"Date"	TEXT NOT NULL,
	FOREIGN KEY("FormId") REFERENCES "forms"("Id") ON UPDATE CASCADE ON DELETE CASCADE,
	PRIMARY KEY("Id" AUTOINCREMENT)
);
